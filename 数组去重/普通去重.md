普通去重

### 基础方法

```
var arr = [2, "1", "2", "1", true, false, true, null, null, undefined, undefined];
function handle(arr) {
    var res = [arr[0]];
    for (var i = 1; i < arr.length; i++) {
        var repeat = false;
        for (var j = 0; j < res.length; j++) {
            if (arr[i] === res[j]) {
                repeat = true;
                break;
            }
        }
        if (!repeat) {
            res.push(arr[i]);
        }
    }
    return res;
}
```

### 使用splice() 

```
// splice 会影响原数组
var arr = [2, "1", "2", "1", true, false, true, null, null, undefined, undefined];
function handle(arr) {
    var len = arr.length;
    for (var i = 0; i < len; i++) {
        for (var j = i + 1; j < len; j++) {
            if (arr[i] === arr[j]) {
                arr.splice(j, 1);
                len--;
                j--;
            }
        }
    }
    return arr;
}
```

### 使用对象的键值

```
// 利用键名不重复
var arr = [2, "1", "2", "1", true, false, true, null, null, undefined, undefined];
function handle(arr) {
    var res = [];
    var hash = {};
    for (var i = 0; i < arr.length; i++) {
        var item = arr[i];
        var key = typeof (item) + item;
        if (hash[key] !== 1) {
            res.push(arr[i]);
            hash[key] = 1;
        }
    }
    return res;
}
```

### 使用indexOf()

```
// indexOf IE9+
// indexOf 使用 严格相等 机制
var arr = [2, "1", "2", "1", true, false, true, null, null, undefined, undefined];
function handle(arr) {
    var res = [];
    for (var i = 0; i < arr.length; i++) {
        if (res.indexOf(arr[i]) == -1) {
            res.push(arr[i]);
        }
    }
    return res;
}
```

### 使用filter()

```
// indexOf IE9+ filter IE9+
// filter 创建一个新数组，数组元素为方法里面返回true的项
var arr = [2, "1", "2", "1", true, false, true, null, null, undefined, undefined];
function handle(arr) {
    var res = arr.filter(function (currentValue, index, arr) {
        return arr.indexOf(currentValue) === index; // 返回true，添加到新数组
    });
    return res;
}
```
